async function fetchData(){ let res = await fetch("https://api.binance.com/api/v3/klines?symbol=ETHUSDT&interval=15m&limit=120"); let data = await res.json(); let closes = data.map(c=>parseFloat(c[4])); let highs = data.map(c=>parseFloat(c[2])); let lows = data.map(c=>parseFloat(c[3])); let price = closes[closes.length-1]; document.getElementById("price").innerText = price.toFixed(2); function MA(arr,p){return arr.slice(-p).reduce((a,b)=>a+b)/p;} let ma7 = MA(closes,7); let ma25 = MA(closes,25); let resistance = Math.max(...highs.slice(-20)); let support = Math.min(...lows.slice(-20)); let trend="震荡结构"; let mode="等待"; let entry=""; let sl=""; let tp=""; if(ma7 > ma25) trend="偏多结构"; if(ma7 < ma25) trend="偏空结构"; /* ===== 做多逻辑 ===== */ if(price <= support+5 && ma7>=ma25){ mode="低吸做多"; entry="做多位置："+(support+2).toFixed(2); sl=(support-18).toFixed(2); tp=(support+40).toFixed(2); document.getElementById("mode").className="green"; } /* ===== 做空逻辑 ===== */ else if(price >= resistance-5 && ma7<=ma25){ mode="高抛做空"; entry="做空位置："+(resistance-2).toFixed(2); sl=(resistance+18).toFixed(2); tp=(resistance-40).toFixed(2); document.getElementById("mode").className="red"; } else{ mode="等待靠近边缘"; entry="支撑："+support.toFixed(2)+" / 压力："+resistance.toFixed(2); sl="固定20止损"; tp="30-60波段"; document.getElementById("mode").className="yellow"; } document.getElementById("trend").innerText=trend; document.getElementById("mode").innerText=mode; document.getElementById("entry").innerText=entry; document.getElementById("sl").innerText=sl; document.getElementById("tp").innerText=tp; }
